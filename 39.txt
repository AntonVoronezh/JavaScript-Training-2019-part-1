    <p class="p11">6</p>
    <button class="btn11">llllll</button>


    <script>
        const set11 = new Set();

        const p11 = document.querySelectorAll('.p11');
        const btn11 = document.querySelector('.btn11');

        for (elem of p11) {

            elem.addEventListener('click', function () {

                set11.add(this);

            });

        };

        btn11.addEventListener('click', () => {

            for(elem of set11) {

                elem.innerHTML += '!';

            }

            set11.clear();

        });



    </script>
